CREATE TABLE customer (
	customer_id varchar(20) PRIMARY KEY, 
	Name varchar(30), 
	address varchar(50),
	email varchar(20)
);

Create table store (
	Store_id int,
	Location varchar(50),
	Hours varchar(6),
	Store_manager varchar(50),
	Primary key (store_id)
);

Create table transactions(
	Transaction_id	 int,
	Customer_id varchar(20),
	Total_price numeric(6,2),
	Type varchar(10),
	Payment_type varchar(8),
	Date date,
	Primary key(transaction_id),
	Foreign key (customer_id) references customer (customer_id)
);

Create table order_history(
	Store_id int,
	Customer_id  varchar(20),
	Transaction_id int,
	Primary key (store_id, customer_id, transaction_id),
	Foreign key (store_id) references store(store_id),
	Foreign key (customer_id) references customer (customer_id),
	Foreign key (transaction_id) references transactions ( transaction_id)
);

create table purchased_goods(
Transaction_id int,
product_id numeric(12,0),
product_count int,
product_price numeric(6,2),
primary key(transaction_id, product_id)
);

create table supplier (
supplier_name varchar(20),
primary key (supplier_name)
);

create table warehouse (
warehouse_id numeric(3,0),
location varchar(50),
primary key (warehouse_id)
);

create table products (
product_id numeric(12,0),
product_name varchar(50),
product_stock numeric(3,0),
product_type varchar(20),
primary key (product_id)
);

create table delivers_to (
supplier_name varchar(20),
warehouse_id int,
delivery_date  DATE NOT NULL DEFAULT CURRENT_DATE,  
foreign key (supplier_name) references supplier(supplier_name),
foreign key (warehouse_id) references warehouse(warehouse_id)
);

create table warehouse_products (
warehouse_id numeric(3,0),
product_id numeric(12,0),
foreign key (warehouse_id) references warehouse(warehouse_id),
foreign key (product_id) references products(product_id)
);

create table supplier_product (
supplier_name varchar(20),
product_id numeric(12,0),
foreign key (supplier_name) references supplier(supplier_name),
foreign key (product_id) references products(product_id)
);

Create table role (
id int, 
name varchar(20), 
primary key(id)
);

Create table employee (id int, 
name varchar(50), 
ssn varchar(4), 
salary numeric(10,2),
DoB date, 
address varchar(90), 
contact varchar(10), 
store_id int, role_id int, 
createdAt date, 
updatedAt date, 
primary key (id), 
foreign key (store_id ) references store(store_id), 
foreign key (role_id ) references role(id) 
) ;

Create table manages (
employee_id int, manager_id int, 
foreign key (employee_id ) references employee (id) ,
foreign key (manager_id  ) references employee (id)
);

create table delivers_to_store (
store_id int,
warehouse_id int,
delivery_date  DATE NOT NULL DEFAULT CURRENT_DATE,
foreign key (store_id) references store(store_id),
foreign key (warehouse_id) references warehouse(warehouse_id)
);
